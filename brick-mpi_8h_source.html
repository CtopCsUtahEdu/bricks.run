519  // All Brick Storage are initialized with mmap for exchanging using views
520  assert(bStorage.mmap_info != nullptr);
521 
522  // Stages are exchanges along one axis.
523  std::vector<MultiStageExchangeView::Stage> send, recv;
524 
525  // Generate all neighbor bitset
526  // The skin_list is no longer important however it is useful to compose the messages.
527  std::vector<BitSet> neighbors;
528  allneighbors(0, 1, dim, neighbors);
529 
530  // memfd that backs the canonical view
531  auto memfd = (MEMFD *) bStorage.mmap_info;
532 
533  BitSet exchanged = 0;
534 
535  for (int d = 1; d <= dim; ++d) {
536  BitSet toexchange = exchanged;
537  BitSet exchanging = 0;
538  exchanging.flip(d);
539  exchanging.flip(-d);
540  toexchange = toexchange | exchanging;
541  send.emplace_back();
542  recv.emplace_back();
543  // Exchanging in one direction at a time
544  for (auto &n: neighbors)
545  if (n <= exchanging && n.set != 0) { // This neighbor is the one exchanging with
546  MultiStageExchangeView::Package sendpkg, recvpkg;
547  size_t len = 0;
548  std::vector<size_t> rpacking;
549  std::vector<size_t> spacking;
550  // Receiving the stuff from the current neighbor
551  for (auto g: ghost)
552  if (g.neighbor.set == n.set) {
553  rpacking.push_back(g.pos * bStorage.step * sizeof(bElem));
554  size_t l = g.len * bStorage.step * sizeof(bElem);
555  rpacking.push_back(l);
556  len += l;
557  }

559  for (auto s: skin)

561  spacking.push_back(s.pos * bStorage.step * sizeof(bElem));
562  spacking.push_back(s.len * bStorage.step * sizeof(bElem));
563  }
564  // Receiving from neighbor's neighbor, assuming neighbor is ordered much like our own.
565  BitSet n2 = n | exchanged;
566  for (auto &g: ghost) {
567  if (g.neighbor <= n2 && !(g.neighbor <= exchanged) && !(g.neighbor.set == n.set)) {
568  // this need to be exchanged
569  rpacking.push_back(g.pos * bStorage.step * sizeof(bElem));
570  size_t l = g.len * bStorage.step * sizeof(bElem);
571  rpacking.push_back(l);
572  len += l;
573  // The corresponding us part also needs to be exchanged
574  // Find the corresponding neighbor
575  BitSet s = g.neighbor ^n;
576  for (int sec = g.skin_st; sec < g.skin_ed; ++sec)
577  // looking for the corresponding skin part
578  for (auto g2: ghost)
579  if (g2.neighbor.set == s.set)
580  if (sec >= g2.skin_st && sec < g2.skin_ed) {
581  auto pos = g2.pos;

588  spacking[last - 1] += seclen;
589  else {
590  spacking.push_back(fst);
591  spacking.push_back(seclen);
592  }
593  }
594  }
595  }
596  recvpkg.len = len;
597  recvpkg.buf = memfd->packed_pointer(rpacking);
598  recvpkg.rank = rank_map[n.set];
599  recv.back().emplace_back(recvpkg);
600  // Sending my stuff
601  size_t ll = 0;
602  for (int i = 0; i < spacking.size(); i += 2)
603  ll += spacking[i + 1];

606  sendpkg.rank = rank_map[Sn.set];

608  }

610  }
611  return MultiStageExchangeView(comm, send, recv);

699  mpi_stats ret = {0, 0, 0, 0};
700  if (comm == MPI_COMM_NULL)
701  return ret;
702  int size;
703  MPI_Comm_size(comm, &size);
704  MPI_Reduce(&stats, &ret.avg, 1, MPI_DOUBLE, MPI_SUM, 0, comm);
705  ret.avg = ret.avg;
706  MPI_Reduce(&stats, &ret.max, 1, MPI_DOUBLE, MPI_MAX, 0, comm);
707  MPI_Reduce(&stats, &ret.min, 1, MPI_DOUBLE, MPI_MIN, 0, comm);
708  stats = stats * stats;
709  MPI_Reduce(&stats, &ret.sigma, 1, MPI_DOUBLE, MPI_SUM, 0, comm);
710  ret.sigma -= ret.avg * ret.avg / size;
711  ret.avg /= size;
712  ret.sigma /= std::max(size - 1, 1);
713  ret.sigma = std::sqrt(ret.sigma);
714  return ret;
715 }
